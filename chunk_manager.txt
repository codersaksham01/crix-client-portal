className="brand-sub">{settings.managerProfile?.name||'Manager'}</div></div></div></div><div className="nav-section"><div className="nav-label">Workspace</div><div className={`nav-item className={`nav-item ${page==='clients'&&!selId?'active':''}`} onClick={()=>{setSelId(null);setPage('clients');setMobileMenuOpen(false);}}><Icon n="users" s={16}/><span>Clients</span></div><div className={`nav-item ${page==='notifications'?'active':''}`} onClick={()=>{setSelId(null);setPage('notifications');setMobileMenuOpen(false);}}><Icon n="bell"   useEffect(()=>{if(ref.current)ref.current.scrollTop=ref.current.scrollHeight;},[messages.length]);
  return <div style={{display:'flex',flexDirection:'column',height:'calc(100vh - 220px)',minHeight:280}} className="fade-in">
    <h2 style={{fontFamily:'var(--font-display)',fontWeight:800,fontSize:20,marginBottom:16}}>Messages</h2>

      <div ref={ref} style={{flex:1,overflowY:'auto',padding:16}} className="space-2">
        {messages.length===0?<EmptyState icon="💬" text="No messages yet."/>:
          messages.map(m=><div key={m.id} style={{display:'flex',justifyContent:m.sender==='client'?'flex-end':'flex-start',gap:8,alignItems:'flex-end'}}>
            {m.sender!=='client'&&<Avatar name={m.senderName||settings.businessName} size={28} color='#111111'/>}
            <div>
              <div className={`chat-bubble ${m.sender==='client'?'sent':'recv'}`}>{m.text}</div>
              <AttachmentList attachments={m.attachments}/>
              <div style={{fontSize:10,color:'var(--text3)',marginTop:3,textAlign:m.sender==='client'?'right':'left'}}>{ago(m.ts)}</div>
            </div>
          </div>)}
      </div>
    </div>
  </div>;
}

function PortalBottomNav({page,setPage,setMenu,client}){
  const adminMsgs=(client.messages||[]).filter(m=>m.sender==='admin').length;
  const openTickets=(client.tickets||[]).filter(t=>t.status==='open').length;
  const pendingApprovals=(client.approvals||[]).filter(a=>a.status==='pending').length;
  const items=[
    {id:'dashboard',label:'Home',icon:'dashboard'},
    {id:'messages',label:'Messages',icon:'message',badge:adminMsgs},
    {id:'invoices',label:'Invoices',icon:'invoice'},

  return <div className="portal-bottom-nav">
    {items.map(n=><div key={n.id} className={`portal-bnav-item ${page===n.id?'active':''}`} onClick={()=>setPage(n.id)}>

function ClientPortalApp({clientId}){
  const [client,setClient]=useState(null);
  const [settings,setSettings]=useState(DEF_SETTINGS);
  const [loading,setLoading]=useState(true);
  const [page,setPage]=useState('dashboard');
  const [portalMenu,setPortalMenu]=useState(false);
  const [showMoreNav,setShowMoreNav]=useState(false);
  const {show,ToastEl}=useToast();

  useEffect(()=>{
    const fetchClient = db()
      ? db().collection(FS_COL).doc(clientId).get().then(snap=>snap.exists?normalizeClientData({id:snap.id,...snap.data()}):null).catch(()=>null)
      : Promise.resolve(null);
    Promise.all([getSettings(), fetchClient]).then(([s,clientData])=>{
      setSettings(s);
      if(clientData) setClient(clientData);
      else {
        // Fallback: try localStorage
        const all = (()=>{try{const d=localStorage.getItem('ut_clients_v3');return d?JSON.parse(d):[];}catch{return[];}})();
        const found = all.find(c=>c.id===clientId);
        if(found) setClient(normalizeClientData(found));
      }
      setLoading(false);
    }).catch(()=>{setLoading(false);});
  },[clientId]);

  // Real-time updates for client
  useEffect(()=>{
    if(!clientId||!db())return;
    const unsub=db().collection(FS_COL).doc(clientId).onSnapshot(snap=>{
      if(snap.exists)setClient(normalizeClientData({id:snap.id,...snap.data()}));
    },()=>{});
    return()=>unsub();
  },[clientId]);

  const updClient=fn=>{
    const updated=normalizeClientData(fn(client));
    setClient(updated);
    saveClient(updated);
  };

  if(loading)return<Loader/>;
  if(!client)return <div className="login-page">
    <div className="login-card" style={{textAlign:'center'}}>
      <div style={{fontSize:48,marginBottom:12}}>🔍</div>
      <h2 style={{fontFamily:'var(--font-display)',fontWeight:800}}>Portal Not Found</h2>
      <p style={{color:'var(--text3)',marginTop:8}}>This client portal link is invalid or has expired.</p>
    </div>
  </div>;

  const unread=(client.notifications||[]).filter(n=>!n.read).length;
  const markRead=()=>{updClient(c=>({...c,notifications:(c.notifications||[]).map(n=>({...n,read:true}))}));};

  const primaryNav=[
    {id:'dashboard',label:'Dashboard',icon:'dashboard'},
    {id:'messages',label:'Messages',icon:'message'},
    {id:'invoices',label:'Invoices',icon:'invoice'},
    {id:'project',label:'My Project',icon:'project'},
    {id:'files',label:'Files',icon:'files'},
    {id:'approvals',label:'Approvals',icon:'check'},
    ...(client.websiteChangeRequestEnabled?[{id:'requests',label:'Website Changes',icon:'support'}]:[]),
    {id:'tickets',label:'Support',icon:'support'},
  ];
  const secondaryNav=[
    {id:'proposals',label:'Proposals',icon:'note'},
    {id:'contracts',label:'Contracts',icon:'contract'},
    {id:'announcements',label:'Announcements',icon:'bell'},
    {id:'feedback',label:'Feedback',icon:'feedback'},
    {id:'activity',label:'Activity',icon:'analytics'},
    {id:'onboarding',label:'Onboarding',icon:'onboard'},
  ];
  const activeInSecondary=secondaryNav.some(n=>n.id===page);

  return <div className="portal-page" style={{minHeight:'100vh',display:'flex',flexDirection:'column'}}>
    {/* Header */}
    <header className="portal-header">
      <div style={{display:'flex',alignItems:'center',gap:10}}>
        <button className="btn btn-ghost btn-icon" onClick={()=>setPortalMenu(!portalMenu)} style={{flexShrink:0}}><Icon n="menu" s={18}/></button>
        <div style={{width:32,height:32,borderRadius:8,overflow:'hidden',flexShrink:0}}>
          <img src={LOGO_B64} alt="Logo" style={{width:'100%',height:'100%',objectFit:'contain'}}/>
        </div>
        <div style={{minWidth:0}}>
          <div style={{fontFamily:'var(--font-display)',fontWeight:700,fontSize:14,color:'var(--text)',overflow:'hidden',textOverflow:'ellipsis',whiteSpace:'nowrap'}}>{settings.businessName}</div>
        </div>
      </div>
      <div style={{display:'flex',alignItems:'center',gap:8}}>
        {unread>0&&<button className="btn btn-ghost btn-icon" onClick={markRead} style={{position:'relative'}}>
          <Icon n="bell" s={16}/>
          <span style={{position:'absolute',top:2,right:2,width:16,height:16,background:'#ef4444',borderRadius:'50%',fontSize:9,color:'#fff',display:'flex',alignItems:'center',justifyContent:'center',fontWeight:700}}>{unread}</span>
        </button>}
        <Avatar name={client.name} size={30} color={settings.brandColor||'#f8fafc'}/>
      </div>
    </header>

    {/* Mobile menu overlay */}
    {portalMenu&&<div style={{position:'fixed',inset:0,background:'rgba(0,0,0,.6)',zIndex:99,backdropFilter:'blur(2px)'}} onClick={()=>setPortalMenu(false)}/>}

    {/* Slide-in sidebar menu */}
    <div style={{position:'fixed',top:0,left:0,bottom:0,width:260,background:'var(--bg2)',borderRight:'1px solid var(--border)',zIndex:100,transform:portalMenu?'translateX(0)':'translateX(-100%)',transition:'transform .25s ease',display:'flex',flexDirection:'column'}}>

            <div style={{fontFamily:'var(--font-display)',fontWeight:700,fontSize:14}}>{client.name}</div>

/* ── PORTAL: FEEDBACK ── */
function PortalFeedback({client,updClient,show}){
  const feedback=(client.feedback||[]).filter(f=>f.response===null);
  const answered=(client.feedback||[]).filter(f=>f.response!==null);
  const respond=(id,response)=>{updClient(c=>addLog({...c,feedback:(c.feedback||[]).map(f=>f.id===id?{...f,response}:f)},'Feedback submitted'));show('Feedback submitted!');};
  const [ratings,setRatings]=useState({});const [texts,setTexts]=useState({});
  return <div className="space-4 fade-in">
    <h2 style={{fontFamily:'var(--font-display)',fontWeight:800,fontSize:20}}>Feedback</h2>
    {feedback.length===0&&answered.length===0?<EmptyState icon="⭐" text="No feedback requests yet."/>:null}
    {feedback.length>0&&<div className="space-3">
      <h3 style={{fontFamily:'var(--font-display)',fontWeight:700,fontSize:15}}>Pending</h3>
      {feedback.map(f=><div key={f.id} className="card card-sm">
        <div style={{fontWeight:700,fontSize:14,marginBottom:10}}>{f.question}</div>
        {f.type==='rating'&&<div style={{display:'flex',gap:4,marginBottom:8}}>{[1,2,3,4,5].map(s=><span key={s} style={{fontSize:28,cursor:'pointer',color:s<=(ratings[f.id]||0)?'#fbbf24':'var(--text3)',transition:'.1s'}} onClick={()=>setRatings(p=>({...p,[f.id]:s}))}>★</span>)}</div>}
        {f.type==='text'&&<textarea className="input" rows="2" placeholder="Your response…" value={texts[f.id]||''} onChange={e=>setTexts(p=>({...p,[f.id]:e.target.value}))} style={{marginBottom:8}}/>}
        {f.type==='yesno'&&<div style={{display:'flex',gap:8,marginBottom:8}}><button className="btn btn-success btn-sm" onClick={()=>respond(f.id,'Yes')}>Yes</button><button className="btn btn-danger btn-sm" onClick={()=>respond(f.id,'No')}>No</button></div>}
        {f.type!=='yesno'&&<button className="btn btn-primary btn-sm" onClick={()=>{const val=f.type==='rating'?ratings[f.id]:texts[f.id];if(val)respond(f.id,val);}}>Submit</button>}
      </div>)}
    </div>}
    {answered.length>0&&<div className="space-3">
      <h3 style={{fontFamily:'var(--font-display)',fontWeight:700,fontSize:15}}>Completed</h3>
      {answered.map(f=><div key={f.id} className="card card-sm" style={{opacity:.7}}>
        <div style={{fontWeight:600,fontSize:13}}>{f.question}</div>
        <div style={{marginTop:6,fontSize:13,color:'var(--text2)'}}>Your response: {f.type==='rating'?'★'.repeat(f.response):String(f.response)}</div>
      </div>)}
    </div>}
  </div>;
}
