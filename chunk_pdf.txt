function buildPDFHeader(){
  return `<div class="header"><div class="header-left"><img src="${LOGO_B64}" alt="Logo" style="width:44px;height:44px;border-radius:10px;object-fit:contain"><div><div class="company-name">${COMPANY.name}</div><div class="company-tag">${COMPANY.tagline}</div></div></div><div class="header-right">${COMPANY.phone}<br>${COMPANY.email}<br>${COMPANY.website}</div></div><div class="accent-bar"></div>`;
}

function buildPDFFooter(){
  return `<div class="footer">${COMPANY.name} &nbsp;|&nbsp; ${COMPANY.phone} &nbsp;|&nbsp; ${COMPANY.email}</div>`;
}

function downloadReportPDF(title,subtitle,headers,rows,summaryCards){
  let html=buildPDFHeader();
  html+=`<div class="page">`;
  html+=`<div class="doc-title">${title}</div>`;
  if(subtitle)html+=`<div class="doc-subtitle">${subtitle}</div>`;
  html+=`<div class="doc-date">Generated: ${new Date().toLocaleDateString('en-IN',{day:'numeric',month:'long',year:'numeric'})}</div>`;
  html+=`<div class="divider"></div>`;
  if(summaryCards&&summaryCards.length){
    html+=`<div class="summary-row">${summaryCards.map(c=>`<div class="summary-card"><div class="summary-label">${c.label}</div><div class="summary-value">${c.value}</div></div>`).join('')}</div>`;
  }
  if(headers&&rows){
    html+=`<table><thead><tr>${headers.map(h=>`<th>${h}</th>`).join('')}</tr></thead><tbody>${rows.map(r=>`<tr>${r.map(c=>`<td>${c||'—'}</td>`).join('')}</tr>`).join('')}</tbody></table>`;
  }
  html+=`</div>`;
  html+=buildPDFFooter();
  openPrintPDF(html,title);
}

function downloadInvoicePDF(inv,client,settings){
  let html=buildPDFHeader();
  html+=`<div class="page">`;
  html+=`<div class="doc-title">INVOICE</div>`;
  html+=`<div class="doc-date">Generated: ${new Date().toLocaleDateString('en-IN',{day:'numeric',month:'long',year:'numeric'})}</div>`;
  html+=`<div class="divider"></div>`;
  // Client + Invoice info side by side
  html+=`<div style="display:flex;gap:16px;margin-bottom:20px;flex-wrap:wrap">`;
  html+=`<div class="client-box" style="flex:1;min-width:200px"><div class="client-label">Bill To</div><div class="client-name">${client.name||client.company||''}</div>`;
  if(client.company)html+=`<div class="client-detail">${client.company}</div>`;
  if(client.email)html+=`<div class="client-detail">${client.email}</div>`;
  if(client.phone)html+=`<div class="client-detail">${client.phone}</div>`;
  html+=`</div>`;
  const badgeCls=inv.status==='paid'?'badge-paid':inv.status==='overdue'?'badge-overdue':inv.status==='sent'?'badge-sent':'badge-draft';
  html+=`<div class="client-box" style="min-width:180px"><div class="client-label">Invoice Details</div><div class="inv-num">${inv.number||'—'}</div>`;
  html+=`<div class="inv-detail">Date: ${inv.createdAt?new Date(inv.createdAt).toLocaleDateString('en-IN',{day:'numeric',month:'short',year:'numeric'}):'—'}</div>`;
  html+=`<div class="inv-detail">Due: ${inv.dueDate?new Date(inv.dueDate).toLocaleDateString('en-IN',{day:'numeric',month:'short',year:'numeric'}):'—'}</div>`;
  html+=`<div class="inv-badge ${badgeCls}">${(inv.status||'draft').toUpperCase()}</div>`;
  html+=`</div></div>`;
  // Table
  html+=`<table><thead><tr><th>Description</th><th style="text-align:right">Amount</th></tr></thead><tbody><tr><td>${inv.description||''}</td><td style="text-align:right;font-weight:700">₹${Number(inv.amount||0).toLocaleString('en-IN')}</td></tr></tbody></table>`;
  // Totals
  const gstRate=inv.gstRate||0;
  const gstAmt=inv.gstAmount||Math.round((inv.amount||0)*gstRate/100);
  const total=(inv.amount||0)+gstAmt;
  html+=`<div class="total-section">`;
  html+=`<div class="total-row"><span class="total-label">Subtotal</span><span>₹${Number(inv.amount||0).toLocaleString('en-IN')}</span></div>`;
  if(gstRate>0){
    html+=`<div class="total-row"><span class="total-label">GST (${gstRate}%)</span><span>₹${gstAmt.toLocaleString('en-IN')}</span></div>`;
  }
  html+=`<div class="total-row grand"><span class="total-label">TOTAL</span><span>₹${total.toLocaleString('en-IN')}</span></div>`;
  html+=`</div>`;
  if(inv.notes){
    html+=`<div class="notes-section"><div class="notes-title">Notes</div><div class="notes-text">${inv.notes}</div></div>`;
  }
  html+=`</div>`;
  html+=buildPDFFooter();
  openPrintPDF(html,inv.number||'Invoice');
}

const escHtml=s=>String(s??'').replace(/&/g,'&amp;').replace(/</g,'&lt;').replace(/>/g,'&gt;');
const pdfCurr=n=>`${(n??0).toLocaleString('en-IN')}`;
const pdfSection=(title,body)=>`<div style="margin-top:20px"><div style="font-size:10px;text-transform:uppercase;font-weight:800;letter-spacing:.14em;color:#64748b;margin-bottom:8px">${escHtml(title)}</div>${body}</div>`;
const pdfList=items=>`<div style="display:grid;gap:8px">${items.map(i=>`<div style="padding:10px 12px;border:1px solid #e5e7eb;border-radius:10px;background:#fff">${i}</div>`).join('')}</div>`;
const pdfSummaryCards=cards=>`<div class="summary-row">${cards.map(c=>`<div class="summary-card"><div class="summary-label">${escHtml(c.label)}</div><div class="summary-value">${escHtml(c.value)}</div></div>`).join('')}</div>`;
const pdfStatusChip=(label,color='#111111')=>`<span style="display:inline-block;padding:4px 10px;border-radius:999px;background:${color};color:#fff;font-size:10px;font-weight:800;letter-spacing:.05em;text-transform:uppercase">${escHtml(label)}</span>`;
function openBrandedPDF({title,subtitle='',coverTitle='',watermark='',body='',filename='document'}){
  let html=buildPDFHeader();
  html+=`<div class="page">`;
  if(coverTitle){
    html+=`<div style="padding:42px 0 28px;border-bottom:1px solid #e5e7eb;margin-bottom:22px">
      <div style="font-size:11px;text-transform:uppercase;letter-spacing:.22em;color:#64748b;font-weight:800;margin-bottom:10px">${escHtml(COMPANY.name)}</div>
      <div style="font-size:34px;font-weight:900;line-height:1.05;color:#0f1117;max-width:560px">${escHtml(coverTitle)}</div>
      ${subtitle?`<div style="font-size:14px;color:#64748b;margin-top:12px;max-width:560px">${escHtml(subtitle)}</div>`:''}
      <div style="font-size:11px;color:#94a3b8;margin-top:20px">Generated ${new Date().toLocaleDateString('en-IN',{day:'numeric',month:'long',year:'numeric'})}</div>
    </div>`;
  }else{
    html+=`<div class="doc-title">${escHtml(title)}</div>`;
    if(subtitle)html+=`<div class="doc-subtitle">${escHtml(subtitle)}</div>`;
    html+=`<div class="doc-date">Generated: ${new Date().toLocaleDateString('en-IN',{day:'numeric',month:'long',year:'numeric'})}</div><div class="divider"></div>`;
  }
  if(watermark)html+=`<div style="position:fixed;top:45%;left:50%;transform:translate(-50%,-50%) rotate(-24deg);font-size:84px;font-weight:900;color:rgba(255,255,255,.08);pointer-events:none;z-index:0">${escHtml(watermark)}</div>`;
  html+=`<div style="position:relative;z-index:1">${body}</div></div>`;
  html+=buildPDFFooter();
  openPrintPDF(html,filename||title);
}

function downloadOutstandingPaymentsPDF(clients){
  const rows=[];
  clients.forEach(c=>ensureArr(c.invoices).forEach(inv=>{const balance=getInvoiceBalance(inv); if(balance>0)rows.push([c.name||'Client',inv.number||'—',inv.description||'',`₹${pdfCurr(balance)}`,inv.dueDate?fmt(inv.dueDate):'—',inv.status||'draft']);}));
  downloadReportPDF('Outstanding Payments','Invoices with pending balances',['Client','Invoice','Description','Balance','Due','Status'],rows,[{label:'Open Invoices',value:rows.length},{label:'Total Pending',value:`₹${pdfCurr(rows.reduce((a,r)=>a+Number(String(r[3]).replace(/[^\d.]/g,'')),0))}`}]);
}

function downloadTaxSummaryPDF(clients){
  let totalTax=0,totalInvoiced=0;
  clients.forEach(c=>ensureArr(c.invoices).forEach(inv=>{totalTax+=Number(inv.gstAmount||0);totalInvoiced+=getInvoiceTotal(inv);}));
  downloadReportPDF('Tax Summary','GST / tax overview across all invoices',['Metric','Value'],[['Total Invoiced',`₹${pdfCurr(totalInvoiced)}`],['Total GST/Tax',`₹${pdfCurr(totalTax)}`],['Invoice Count',String(clients.reduce((a,c)=>a+ensureArr(c.invoices).length,0))]],[{label:'Total Tax',value:`₹${pdfCurr(totalTax)}`},{label:'Invoices',value:clients.reduce((a,c)=>a+ensureArr(c.invoices).length,0)}]);
}

function downloadProfitabilityPDF(clients,settings){
  const rows=clients.map(c=>{const revenue=ensureArr(c.invoices).reduce((a,i)=>a+getInvoicePaidAmount(i),0);const expenses=ensureArr(c.expenses).reduce((a,e)=>a+Number(e.amount||0),0);const hours=ensureArr(c.timeEntries).reduce((a,t)=>a+Number(t.hours||0),0);const labor=hours*Number(settings.hourlyRate||0);const profit=revenue-expenses-labor;return [c.name||'Client',`₹${pdfCurr(revenue)}`,`₹${pdfCurr(expenses)}`,`${hours.toFixed(2)}h`,`₹${pdfCurr(profit)}`];});
  downloadReportPDF('Profitability Report','Revenue, expenses, labor, and estimated profit',['Client','Collected Revenue','Expenses','Hours','Estimated Profit'],rows,[{label:'Clients',value:clients.length},{label:'Hourly Rate',value:`₹${pdfCurr(settings.hourlyRate||0)}`}]);
}
